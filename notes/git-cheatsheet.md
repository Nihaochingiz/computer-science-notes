## Создание репозитория

```bash
# Создать новый локальный репозиторий
git init [название проекта]

# Клонировать существующий репозиторий
git clone [url-адрес]
```

## Внесение изменений

```bash
# Проверить статус файлов
git status

# Показать изменения в неиндексированных файлах
git diff

# Добавить файл в индекс
git add [файл]

# Добавить все файлы
git add .

# Показать изменения в индексированных файлах
git diff --staged

# Отменить индексацию файла
git reset [файл]

# Создать коммит
git commit -m "[сообщение]"
```

## Работа с ветками

```bash
# Показать список веток
git branch

# Создать новую ветку
git branch [имя ветки]

# Переключиться на ветку
git checkout [имя ветки]

# Создать и переключиться на ветку
git checkout -b [имя ветки]

# Слить ветку в текущую
git merge [имя ветки]

# Удалить ветку
git branch -d [имя ветки]
```

## Операции с файлами

```bash
# Удалить файл из репозитория
git rm [файл]

# Прекратить отслеживание файла (оставить локально)
git rm --cached [файл]

# Переименовать файл
git mv [старое] [новое]
```

## Игнорирование файлов

Создайте файл `.gitignore`:

```
*.log
build/
temp-*
.DS_Store
node_modules/
```

```bash
# Проверить игнорируемые файлы
git ls-files --others --ignored --exclude-standard
```

## Временные изменения (stash)

```bash
# Сохранить изменения
git stash

# Восстановить и удалить из стеша
git stash pop

# Показать список стешей
git stash list

# Удалить последний стеш
git stash drop
```

## Просмотр истории

```bash
# Показать историю коммитов
git log

# История изменений файла
git log --follow [файл]

# Краткая история (одна строка на коммит)
git log --oneline

# Сравнить две ветки
git diff [ветка1]..[ветка2]

# Показать изменения в коммите
git show [коммит]
```

## Откат изменений

```bash
# Мягкий сброс (сохраняет изменения в рабочей директории)
git reset [коммит]

# Жёсткий сброс (удаляет все изменения)
git reset --hard [коммит]

# Отменить последний коммит (сохраняя изменения)
git reset --soft HEAD~1

# Создать коммит-отмену
git revert [коммит]
```

## Работа с удалённым репозиторием

```bash
# Добавить удалённый репозиторий
git remote add origin [url]

# Показать удалённые репозитории
git remote -v

# Загрузить изменения с сервера
git fetch [remote]

# Слить изменения из удалённой ветки
git merge [remote]/[ветка]

# Загрузить и слить (fetch + merge)
git pull

# Отправить изменения на сервер
git push [remote] [ветка]

# Отправить впервые и установить upstream
git push -u origin [ветка]
```

## Полезные команды

```bash
# Показать текущую ветку
git branch --show-current

# Отменить все локальные изменения
git checkout -- .

# Удалить неотслеживаемые файлы
git clean -fd

# Показать изменения в последнем коммите
git show HEAD
```
